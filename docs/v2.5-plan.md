# Cloghan Tank Monitor V2.5 — Detailed Implementation Plan

## Overview

Separate the monolithic V2 app into a **backend** (system tray + API server) and a **frontend** (Vercel-deployed SPA). The backend runs on the user's PC, captures screenshots, calls the AI, and exposes a REST API. The frontend connects via a stable ngrok tunnel URL + API key authentication.

---

```
┌──────────────────────────┐              ┌──────────────────────────────────┐
│  Frontend (Vercel)       │   HTTPS      │  Backend (System Tray)           │
│  React SPA               │◄───API──────►│  Fastify on 127.0.0.1:3000      │
│  cloghan-tanks.vercel.app│   Bearer key │  Exposed via ngrok tunnel        │

```

## Phase 1: Backend — API Key System

### 1.1 Database Migration

**File: `apps/server/src/db.ts`**

Add `api_keys` table to `initDatabase()`:

```sql
CREATE TABLE IF NOT EXISTS api_keys (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  key TEXT NOT NULL UNIQUE,
  label TEXT,
  permissions TEXT NOT NULL CHECK(permissions IN ('readonly', 'readwrite')),
  created_at TEXT NOT NULL,
  revoked_at TEXT
);
```

**New functions to add:**

```typescript
// Generate a new API key
function generateApiKey(label: string, permissions: 'readonly' | 'readwrite'): string {
  const key = `ctm_live_${crypto.randomBytes(16).toString('hex')}`;
  db.prepare(
    'INSERT INTO api_keys (key, label, permissions, created_at) VALUES (?, ?, ?, ?)'
  ).run(key, label, permissions, new Date().toISOString());
  return key;
}

// Validate an API key (returns null if invalid/revoked)
function validateApiKey(key: string): ApiKey | null {
  return db.prepare(
    'SELECT * FROM api_keys WHERE key = ? AND revoked_at IS NULL'
  ).get(key) as ApiKey | null;
}

// List active API keys
function listApiKeys(): ApiKey[] {
  return db.prepare(
    'SELECT id, label, permissions, created_at FROM api_keys WHERE revoked_at IS NULL'
  ).all() as ApiKey[];
}

// Revoke an API key
function revokeApiKey(id: number): void {
  db.prepare('UPDATE api_keys SET revoked_at = ? WHERE id = ?')
    .run(new Date().toISOString(), id);
}
```

### 1.2 Auth Middleware

**New file: `apps/server/src/auth.ts`**

```typescript
import { FastifyRequest, FastifyReply } from 'fastify';

// Cookie-based session auth
export async function authenticate(request: FastifyRequest, reply: FastifyReply) {
  // Check session cookie first
  const session = request.cookies.ctm_session;
  if (session) {
    const keyData = validateApiKey(session);
    if (keyData) {
      request.apiKey = keyData;
      return;
    }
  }

  // Fall back to Authorization header
  const authHeader = request.headers.authorization;
  if (authHeader?.startsWith('Bearer ')) {
    const key = authHeader.slice(7);
    const keyData = validateApiKey(key);
    if (keyData) {
      request.apiKey = keyData;
      return;
    }
  }

  return reply.code(401).send({ error: 'Invalid or missing API key' });
}

// Middleware that requires read-write permissions
export async function requireReadWrite(request: FastifyRequest, reply: FastifyReply) {
  await authenticate(request, reply);
  if (reply.sent) return;
  if (request.apiKey?.permissions !== 'readwrite') {
    return reply.code(403).send({ error: 'Read-write access required' });
  }
}
```

### 1.3 Session Endpoints

**Add to `apps/server/src/index.ts`:**

```typescript
// Login — validates key, sets session cookie
fastify.post('/api/auth', async (request, reply) => {
  const { key } = request.body as { key: string };
  const keyData = validateApiKey(key);
  if (!keyData) {
    return reply.code(401).send({ error: 'Invalid API key' });
  }
  reply.setCookie('ctm_session', key, {
    path: '/',
    httpOnly: true,
    secure: true,      // HTTPS via ngrok
    sameSite: 'none',  // cross-origin from Vercel
    maxAge: 30 * 24 * 60 * 60, // 30 days
  });
  return { ok: true, permissions: keyData.permissions };
});

// Logout — clears session cookie
fastify.delete('/api/auth', async (request, reply) => {
  reply.clearCookie('ctm_session', { path: '/' });
  return { ok: true };
});

// Check current session
fastify.get('/api/auth', async (request, reply) => {
  const session = request.cookies.ctm_session;
  if (session) {
    const keyData = validateApiKey(session);
    if (keyData) {
      return { authenticated: true, permissions: keyData.permissions };
    }
  }
  return { authenticated: false };
});
```

**New file: `apps/server/src/tray.ts`**

- Parse ngrok URL from its local API (port 4040)
- Provide key generation via tray menu
- Show notifications on key generation

**Tray menu structure:**

```
  Cloghan Tank Monitor
  ───────────────────
  ● Server: running on :3000
  🔗 Tunnel: https://xxxx.ngrok-free.app
  ───────────────────
  Generate Read-Only API Key    → generates key, copies to clipboard, notification
  Generate Read/Write API Key   → generates key, copies to clipboard, notification
  View Active API Keys          → opens local key viewer window
  ───────────────────
  Copy Tunnel URL               → copies ngrok URL to clipboard
  ───────────────────
  Pause Capture                 → toggles scheduler
  ───────────────────
  Quit                          → stops ngrok, stops server, exits
```

**Ngrok management:**

```typescript
import { spawn, ChildProcess } from 'child_process';

let ngrokProcess: ChildProcess | null = null;
let ngrokUrl: string | null = null;

function startNgrok() {
  ngrokProcess = spawn('ngrok', ['http', '3000'], {
    stdio: ['ignore', 'pipe', 'pipe'],
  });

  // Parse URL from stdout (ngrok outputs it)
  ngrokProcess.stdout?.on('data', (data) => {
    const match = data.toString().match(/https:\/\/[a-z0-9-]+\.ngrok-free\.app/);
    if (match) {
      ngrokUrl = match[0];
      updateTrayMenu();
    }
  });

  // Also poll ngrok's local API
  const pollUrl = async () => {
    try {
      const res = await fetch('http://127.0.0.1:4040/api/tunnels');
      const data = await res.json();
      ngrokUrl = data.tunnels[0]?.public_url ?? null;
      updateTrayMenu();
    } catch {}
  };
  setInterval(pollUrl, 2000);
}

function stopNgrok() {
  ngrokProcess?.kill();
  ngrokProcess = null;
  ngrokUrl = null;
}
```

## Phase 4: Frontend — Login & Auth

### 4.1 API Helper

**New file: `apps/web/src/api.ts`**

```typescript
const API_BASE = () => localStorage.getItem('serverUrl') ?? '';
const API_KEY = () => localStorage.getItem('apiKey') ?? '';

export function setServerConfig(url: string, key: string) {
  localStorage.setItem('serverUrl', url);
  localStorage.setItem('apiKey', key);
}

export function clearServerConfig() {
  localStorage.removeItem('serverUrl');
  localStorage.removeItem('apiKey');
  document.cookie = 'ctm_session=; path=/; max-age=0';
}

export async function apiRequest<T = unknown>(
  method: string,
  path: string,
  body?: unknown,
): Promise<T> {
  const res = await fetch(`${API_BASE()}${path}`, {
    method,
    headers: {
      ...(body ? { 'Content-Type': 'application/json' } : {}),
      'Authorization': `Bearer ${API_KEY()}`,
    },
    body: body ? JSON.stringify(body) : undefined,
    credentials: 'include',  // Send cookies
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: res.statusText }));
    throw new Error(err.error ?? res.statusText);
  }
  return res.json();
}

export async function testConnection(url: string, key: string): Promise<boolean> {
  try {
    const res = await fetch(`${url}/api/health`);
    return res.ok;
  } catch {
    return false;
  }
}

export async function login(url: string, key: string): Promise<{ permissions: string }> {
  const res = await fetch(`${url}/api/auth`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ key }),
    credentials: 'include',
  });
  if (!res.ok) throw new Error('Invalid API key');
  return res.json();
}

export async function checkSession(url: string): Promise<{ authenticated: boolean; permissions?: string }> {
  const res = await fetch(`${url}/api/auth`, { credentials: 'include' });
  return res.json();
}

export async function logout(url: string) {
  await fetch(`${url}/api/auth`, {
    method: 'DELETE',
    credentials: 'include',
  });
  clearServerConfig();
}
```

### 4.2 Login Screen

**Add to `apps/web/src/App.tsx`:**

```tsx
function LoginScreen({ onLogin }: { onLogin: (url: string, key: string, permissions: string) => void }) {
  const [serverUrl, setServerUrl] = useState('');
  const [apiKey, setApiKey] = useState('');
  const [error, setError] = useState('');
  const [loading, setLoading] = useState(false);

  const handleConnect = async () => {
    setLoading(true);
    setError('');
    try {
      const reachable = await testConnection(serverUrl, apiKey);
      if (!reachable) throw new Error('Cannot reach server');
      const { permissions } = await login(serverUrl, apiKey);
      setServerConfig(serverUrl, apiKey);
      onLogin(serverUrl, apiKey, permissions);
    } catch (e: any) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="login-screen">
      <div className="login-card">
        <h1>Cloghan Tank Monitor</h1>
        <p>Connect to your server</p>
        <div className="login-field">
          <label>Server URL</label>
          <input value={serverUrl} onChange={e => setServerUrl(e.target.value)}
            placeholder="https://xxxx.ngrok-free.app" />
        </div>
        <div className="login-field">
          <label>API Key</label>
          <input value={apiKey} onChange={e => setApiKey(e.target.value)}
            placeholder="ctm_live_..." type="password" />
        </div>
        {error && <div className="login-error">{error}</div>}
        <button className="primary" onClick={handleConnect} disabled={loading}>
          {loading ? 'Connecting...' : 'Connect'}
        </button>
      </div>
    </div>
  );
}
```

### 4.3 Permissions-Aware UI

**Add to `apps/web/src/App.tsx`:**

```tsx
function App() {
  const [permissions, setPermissions] = useState<string | null>(null);
  const [serverUrl, setServerUrl] = useState<string | null>(null);
  const isReadOnly = permissions === 'readonly';

  // On mount: check session
  useEffect(() => {
    const url = localStorage.getItem('serverUrl');
    const key = localStorage.getItem('apiKey');
    if (url && key) {
      checkSession(url).then(s => {
        if (s.authenticated) {
          setServerUrl(url);
          setPermissions(s.permissions ?? 'readonly');
        }
      });
    }
  }, []);

  if (!serverUrl || !permissions) {
    return <LoginScreen onLogin={(url, key, perm) => {
      setServerUrl(url);
      setPermissions(perm);
    }} />;
  }

  return (
    <div className="app">
      <Header
        isReadOnly={isReadOnly}
        onLogout={() => { logout(serverUrl); setServerUrl(null); setPermissions(null); }}
      />
      {/* ... rest of app */}
    </div>
  );
}
```
